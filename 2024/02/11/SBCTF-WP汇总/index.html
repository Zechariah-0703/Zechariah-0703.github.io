#include <stdio.h>
#include <stdint.h>
//加密函数
void encrypt(unsigned int num_rounds, uint32_t *v, int32_t *k)
{
  int32_t v0 = v[0], v1 = v[1], sum = 0, i;
  int32_t delta = 0x9E3779B9;
  int32_t k0 = k[0], k1 = k[1], k2 = k[2], k3 = k[3];
  for (i = 0; i < num_rounds; i++)
  {
     sum += delta;
     v0 += ((v1 << 4) + k0) ^ (v1 + sum) ^ ((v1 >> 5) + k1);
     v1 += ((v0 << 4) + k2) ^ (v0 + sum) ^ ((v0 >> 5) + k3);
  }
  v[0] = v0;
  v[1] = v1;

}
//解密函数
void decrypt(unsigned int num_rounds, uint32_t *v, int32_t *k)
{
  int32_t v0 = v[0], v1 = v[1], i;
  int32_t delta = 0x9E3779B9, sum = delta * num_rounds;
  int32_t k0 = k[0], k1 = k[1], k2 = k[2], k3 = k[3];
  for (i = 0; i < num_rounds; i++)
  {
     v1 -= ((v0 << 4) + k2) ^ (v0 + sum) ^ ((v0 >> 5) + k3);
     v0 -= ((v1 << 4) + k0) ^ (v1 + sum) ^ ((v1 >> 5) + k1);
     sum -= delta;
  }
  v[0] = v0;
  v[1] = v1;
}

ez_block

最自学成才的一集，从22号考完高数后看到这道题开始，到24号近24:00,我基本掌握了AES-CBC的加密流程，使我能够看懂这道题并得到flag，最喜欢学密码的时刻。

这道题的flag是加密流程中的IV

不妨先观察所给的cipher，解两次base64,得到5436d43a1658e7************************bb2cdabc02a058f8c3936a5fd6ddd3c50491

再观察encrypt.py，key = “3N7g309d6Y7enT**”

Cipher残缺，key残缺

基础知识:AES常用的块长度是128bit，通过计算与观察，该题的明文与密文可分为4组，且第4组的密文完整，前三组的密文各泄露了几个字符。

综上所述，我们可以利用完整的第4组密文和残缺的第3组密文(←用于验证)来暴破出密钥

得到密钥后，写一个for循环一组一组进行还原即可还原出IV,即我们所需要的flag

我的完整exp:

import string
from Crypto.Cipher import AES
import binascii
import hashlib

#密钥字典生成
with open("keytable.txt","wb+") as file :
    for i in string.printable[:62]:
        for j in string.printable[:62]:
            key = "3N7g309d6Y7enT"
            key+=i
            key+=j
            file.write(bytes(key.encode()))
            file.write(b"\n")

#异或
def xor(m: bytes, c: bytes): 
    return bytes([i ^ j for i, j in zip(m, c)])

#爆破确认密钥
file = open("keytable.txt","rb+")
enc = b"dabc02a058f8c3936a5fd6ddd3c50491"
three_part = binascii.unhexlify(b"bb2c")
for i in file:
    key = i[:16]
    aes = AES.new(key, AES.MODE_ECB)
    dec = aes.decrypt(binascii.unhexlify(enc))
    four = b"curity is a myth"
    xor_result = xor(four,dec)
    if three_part in xor_result:
        print(binascii.hexlify(xor_result))
        print(key)
        print(xor_result)
        break
file.close()

#还原IV
message = b'Security is not a joke, mind it. But complete security is a myth'
for i in range(int(len(message)/16)):
    m=message[len(message)-16*(i+1):len(message)-16*i]
    aes = AES.new(key, AES.MODE_ECB)
    dec = aes.decrypt(binascii.unhexlify(enc))
    xor_result = xor(m,dec)
    print(binascii.hexlify(xor_result))
    print(xor_result)
    enc = binascii.hexlify(xor_result)

ez_brainfuzz

先阅读题目描述:你知道brainfuck吗？你会fuzz吗？
我们知道，这道题中的brainfuck是一种类似于密码的东西
nc连上去后，要输入，会返回给你similarity的值，猜测是要暴破，直到similarity到100%
再阅读附件中的app.py,可知我们要输入flag的md5的brainfuck形式

经测试，可以一个字符一个字符的确认正确答案。(基础知识环节，md5通常32位)

使用pwntools连接，利用github上下载的string_to_bf函数将以下操作重复32次，即可获得similarity=100%的brainfuck

a="0123456789abcdef"
b=''
presimi=0
for i in a:
     md5=(string_to_bf(b+i,0)
     r.sendline(md5.encode())
     simi=r.recvline().decode().split(" ")[3].split("%")[0]
     if int(simi)>presimi:
       presimi=int(simi)
       print(simi)
       print(b+i)
       b=b+i
       break

sendline过去再recvline即可获得flag

先看加密流程：对于每个像素点，20个在0~1的随机数，进行1次或2次dct，最后归一化处理，使数据在[0,1]之间，取出后19个数据写入out.txt.文件中有26011个数据，26011/19=37*37，可知二维码的规格。

本题难点在于如何通过19个数据，推断出经过了几次dct，从而根据dct次数判断像素点的黑白。

群友提供了看斜率的思路，于是我自己纯手工写了个exp，出来一个二维码，稍微补一补定位块，真能扫出来，win！。

Crypto

ez_pack(位数太低造成非预期🤣)

背包问题？但是由于出题人的疏忽，暴力分解n很快，出现了非预期

分解完n，那么p，q，e，c，d要素齐全，属于是点击就送的RSA。

ez_pack_revenge（遗憾离场）

上一题的复仇题。我认为也是背包问题。但数字太大，算法的方式得不出结果。

加密流程对初学者的我来说略显复杂，但不难看懂，只要根据a中的数据还原出d的二进制表示的每一位上是1 or 0，有了d，问题自然迎刃而解。可是，我一开始觉得这和力扣上的类似背包问题的40.组合总和II是一个类型

着重看了背包算法和剪枝，然后脚本跑了将近6个小时也没跑出结果。

后来群里的一句用：

算法题的方法解256位的子集和问题估计得用行星计算机跑到明年去

破防了，遗憾离场

ez_Signature(很令人感动的flag 😭，附完结感言)

关键在于如何求d

根据出题人的提示：可以爆破X1，X2前2位，然后根据d*G是否等于public判断d是否正确。

from Crypto.Util.number import *
p = 0xfffffffffffffffffffffffffffffffffffffffffffffffffffffffefffffc2f
a = 0
b = 7
E = EllipticCurve(GF(p), [a, b])
G = E.gens()[0]
order = E.order()
public = E(5493730386557588740213129802259912244320687664566838908933139695297485450984, 112890332916025956487764670868844990362012495830331005756101957669156593225707)
s = 56275471718384662997548252167133434300653095757824127552821000258065614147953
r = 88763903751156737929684129280869308730236267295398468396864060901397400254458
msg = '79e3587b06b1caec3323a5f3b944b4946e06aedca38e2b0d6b231c4577a192bf'
z=bytes_to_long(bytes.fromhex(msg))
a1=2^128*r-s
a2=2^128*s-r
a2_=inverse_mod(a2,order)
z=z*a2_
for u in range(4):
    for v in range(4):
        Z=z+a1*a2_*u*2^126-v*2^126
        m=matrix(ZZ,[[Z,0,2^126],
                     [a1*a2_,1,0],
                     [order,0,0]])
        l=m.LLL()
        for i in l:
            if abs(i[-1])==2^126:
                dl=abs(i[0])
                dh=abs(i[1])
                d=(u*2^126+dh)*2^128+v*2^126+dl
                if d*G==public:
                    print('d=',d)
#d=114848009668899858113932463053239179798180491260873213931636897786613068143546
